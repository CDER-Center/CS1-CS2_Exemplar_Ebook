    public void remove(T toRemove) {
        //remove first occurrence of toRemove from the list
        Node prev = null;
        Node curr = head;
        if(curr == null)  //special case: empty list
            return;
        if(curr.value.equals(toRemove)) {  //special case: removing first value
            head = curr.next;
            return;
        }
        while(curr != null) {
            if(curr.value.equals(toRemove)) {
                prev.next = curr.next;
                return;
            }
            prev = curr;
            curr = curr.next;
        }
    }

    public void remove(T toRemove) {
        Node prev = head;
        if(head == null) //special case: empty list
            return;
        if(prev.value.equals(toRemove)) { //special case: removing first value
            head = head.next;
            return;    
        }
        while(prev.next != null) {
            if(prev.next.value.equals(toRemove)) {
                prev.next = prev.next.next;
                return;
            }
            prev = prev.next;
        }
    }
