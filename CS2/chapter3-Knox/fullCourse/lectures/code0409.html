  public boolean contains(T s) {
    //returns whether the list contains value s

    boolean flag = false;  //whether we've seen s
    Node curr = head;
    while(curr != null) {
      if(curr.value.equals(s))
        flag = true;
      curr = curr.next;
    }
    return flag;
  }
